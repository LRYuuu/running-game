### 3.3 实现细节

#### 3.3.1 地面检测逻辑
```csharp
/// <summary>
/// 检测玩家是否在地面上
/// </summary>
/// <returns>True 如果玩家在地面上，否则 False</returns>
public bool IsGrounded()
{
    // 从玩家底部向下发射 Raycast
    var hit = Physics2D.Raycast(
        origin: transform.position,
        direction: Vector2.down,
        distance: _groundCheckDistance,
        layerMask: _groundLayer
    );

    // 可选：Debug.DrawRay 用于可视化检测
    return hit.collider != null;
}
```

#### 3.3.2 跳跃执行逻辑
```csharp
/// <summary>
/// 执行跳跃（仅当在地面时）
/// </summary>
public void Jump()
{
    if (!IsGrounded())
        return;

    // 清除垂直速度，确保跳跃高度一致
    var velocity = _rigidbody2D.velocity;
    velocity.y = 0;
    _rigidbody2D.velocity = velocity;

    // 应用向上力
    _rigidbody2D.AddForce(Vector2.up * _jumpForce, ForceMode2D.Impulse);

    // 日志输出（调试用）
    Debug.Log($"[PlayerJumpController] Jump executed, force: {_jumpForce}");
}
```

#### 3.3.3 输入处理
```csharp
/// <summary>
/// 处理跳跃输入（每帧调用）
/// </summary>
private void Update()
{
    if (_jumpAction.WasPressedThisFrame())
    {
        Jump();
    }
}
```
